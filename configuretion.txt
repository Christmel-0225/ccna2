task 4: design implementation and configuration

d1:
conf t 
vlan 3
name CHEVRON
vlan 4 
name PETRON
vlan 5
name USEMBASSY
vlan 6
name GOODYEARINC

interface vlan 3
desc SVIFORchevron
no shut
ip address 10.0.32.1 255.255.224.0

interface vlan 4
desc SVIFORPETRON
no shut
ip address 10.0.8.1 255.255.248.0

interface vlan 5
desc SVIFORUSEMBASSY
no shut
ip address 10.0.0.129 255.255.255.128

interface vlan 6
desc SVIFORGOODYEAR
no shut
ip address 10.0.16.1 255.255.240.0

ip dhcp excluded-add 10.0.32.1 10.0.32.100
ip dhcp excluded-add 10.0.8.1 10.0.100
ip dhcp excluded-add 10.0.0.129 10.0.0.139
ip dhcp excluded-add 10.0.16.1 10.0.16.100

ip dhcp pool CHEVRON
network 10.0.32.0 255.255.224.0
default-router 10.0.32.1

ip dhcp pool PETRON
network 10.0.8.0 255.255.248.0
default-router 10.0.8.1

ip dhcp pool USEMBASSY
network 10.0.0.129 255.255.255.128
default-router 10.0.0.129

ip dhcp pool GOODYEARINC
network 10.0.16.0 255.255.240.0
default-router 10.0.16.1
end
